const express = require("express"); const cors = require("cors"); require("./db"); const app = express(); app.use(cors()); app.use(express.json()); app.use("/api/auth", require("./routes/authRoutes")); app.use("/api/services", require("./routes/serviceRoutes")); app.use("/api/feedback", require("./routes/feedbackRoutes")); app.listen(5000, () => console.log("Server running on 5000"));
